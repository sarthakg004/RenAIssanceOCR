httpx==0.28.1

# --- AI / OCR providers ---
google-genai==1.0.0

# --- Document export ---
python-docx==1.1.2
reportlab==4.2.5

# --- Image processing ---
Pillow==11.1.0
opencv-python-headless==4.10.0.84
# numpy 1.24.4: required exactly by paddlex 3.0.0 (Python < 3.12).
# Also satisfies paddlepaddle-gpu>=1.21 and opencv-python-headless>=1.23.5.
numpy==1.24.4

# --- Utilities ---
python-dotenv==1.0.1
psutil==6.1.0

# --- paddlepaddle-gpu runtime deps (installed with --no-deps in Dockerfile) ---
# These are the non-CUDA runtime deps that paddle 3.3.0 would normally pull
# transitively. Torch/FastAPI/pydantic already bring filelock, typing_extensions,
# networkx, numpy, Pillow, protobuf, httpx — the only one missing is opt_einsum.
opt_einsum==3.3.0

# --- Local recognition models (CRNN / TrOCR) ---
torch==2.5.1
# transformers must be <=4.44.x to accept tokenizers 0.19.x, which is
# what paddlex 3.0.0 hard-pins ("tokenizers==0.19.1"). transformers 4.45+
# jumped to tokenizers>=0.20 and is resolution-incompatible with paddlex.
# 4.44.2 supports the same VisionEncoderDecoderModel / TrOCRProcessor
# APIs the recognition service relies on.
transformers==4.44.2
# safetensors must be >=0.6.0 to satisfy paddlepaddle-gpu==3.3.0;
# transformers 4.44.2 only requires >=0.4.1, so 0.6.x works for both.
safetensors==0.6.2

# --- PDF parsing ---
PyMuPDF==1.25.1

# --- Layout-aware GPU detection ---
# Requires: NVIDIA driver >= 560 (CUDA 12.6) on host, --gpus all on docker run.
#
# NOTE: paddlepaddle-gpu is NOT listed here — it is installed as a separate
# --no-deps step in the Dockerfile. Reason: paddlepaddle-gpu==3.3.0 hard-pins
# nvidia-nccl-cu12==2.25.1 (among several other nvidia-*-cu12 ==X.Y.Z pins)
# while no torch release on PyPI pins that exact nccl. Letting pip resolve
# both packages together is impossible under the modern resolver. Both
# frameworks bundle their own CUDA user-space libraries under torch/lib/
# and paddle.libs/ respectively, so the nvidia-*-cu12 site-packages copies
# are redundant at runtime — installing paddle with --no-deps is safe.
paddleocr==3.0.0
